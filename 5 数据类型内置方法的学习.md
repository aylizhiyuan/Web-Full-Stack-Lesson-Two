````javascript
    //调用的时候一定要记住，保证数据类型为Number类型,否则无法正常调用内部函数

    //var num = '100.23456789';
    //错误的做法
    //window.tostring()
    //window.tofixed()
    //正确的方法
    //整数类型转化为字符串
    var result = num.toString();

    //保留小数点几位,再转化为字符串

    //将整个整数类型转化为指数形式
    var result = num.toExponential();
    console.log(result);
````

````javascript
    //tolowercase,touppercase分别是转化大小写的
    str1.toLowerCase();
````

````javascript
//数组
````

````javascript
//this是什么东西
    //无论在哪儿啊，this都是window,window是顶级.

            function box3(){
                alert(this);
            }
            box3();
        }
        box2();
    }
    //box();

    //在事件中，this指向的是发生事件的元素
    document.body.onclick = function(e){
        alert(this);
    }

    //在构造函数中this指向的是构造函数本身
    function base(){
        console.log(this);
        this.name = '李志远';
        this.age = '30';
    }
    var result = new base();

    //在对象当中,this就是对象本身.
    var obj = {
        name:'李志远',
        age:'30',
        run:function(){
            alert(this);
            console.log(this);
        }
    }
    obj.run();

    //实例来理解一下
    var name = '李志远';
    var age = 30;

        alert(this.name);

    var name = '李志远';

    function box(){
        //它是window
        alert(this);
        this.name = '班长';
        this.age = 50;
        alert(this.name);

    var result = new box();//new之后就变成box了


    //对象里面的this指向的是对象本身吧
````
